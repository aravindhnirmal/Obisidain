%{
[[include]] <stdio.h>
%}


statement_list: statement
              | statement_list statement
              ;

statement: expression '\n' { printf("Result: %d\n", $1); }
         ;

expression: expression '+' term { $$ = $1 + $3; }
          | expression '-' term { $$ = $1 - $3; }
          | term                { $$ = $1; }
          ;

term: term '*' factor { $$ = $1 * $3; }
    | term '/' factor { $$ = $1 / $3; }
    | factor           { $$ = $1; }
    ;

factor: NUMBER { $$ = $1; }
       | '(' expression ')' { $$ = $2; }
       ;

%%

int main() {
    yyparse();
    return 0;
}

int yyerror(char *s) {
    printf("Error: %s\n", s);
    return 0;
}
