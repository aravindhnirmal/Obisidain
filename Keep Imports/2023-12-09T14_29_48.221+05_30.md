 def preorder_print_recursive(self):
        self._preorder_print_recursive(self.root)

    def _preorder_print_recursive(self, node):
        if node:
            print(node.data, end=" ")
            self._preorder_print_recursive(node.left)
            self._preorder_print_recursive(node.right)
