def inorder_print(self):
        if self.root:
            self._inorder_print_recursively(self.root)

    def _inorder_print_recursively(self, node):
        if node:
            self._inorder_print_recursively(node.left)
            print(node.data, end=" ")
            self._inorder_print_recursively(node.right)
